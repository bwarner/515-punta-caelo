export const metadata = {
  title: "Wi-Fi Information",
  locale: "en",
  darkBackground: true,
};

export const wifiCredentials = {
  ssid: process.env.WIFI_SSID1
    ? Buffer.from(process.env.WIFI_SSID1).toString("base64")
    : "",
  password: process.env.WIFI_PASSWORD1
    ? Buffer.from(process.env.WIFI_PASSWORD1).toString("base64")
    : "",
};

<AltBackground metadata={metadata}>
<Menubar locale={metadata.locale} title={metadata.title} darkBackground={true} />

<WiFiPanel
  encodedSsid={wifiCredentials.ssid}
  encodedPassword={wifiCredentials.password}
/>

</AltBackground>
